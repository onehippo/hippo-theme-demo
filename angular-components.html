AngularJS components

There is an API reference available for the AngularJS components and services that are part of the Hippo theme.

Tree
●
{{ item.title }} 
<div hippo.theme.tree
     items="treeItems"
     callbacks="callbacks"
     class="hippo-tree ng-cloak"
     selected-item-id="selectedItem.id">
  <a class="btn btn-link btn-xs"
     data-nodrag data-ng-click="toggleItem()"
     data-ng-hide="childNodes().length === 0">
    <span class="fa" data-ng-class="{'fa-caret-down': !collapsed, 'fa-caret-right': collapsed}"></span>
  </a>
  <div class="spacer" data-ng-hide="childNodes().length !== 0">●</div>
  {{ item.title }}
</div>

Note: toggleItem() will keep the collapsed states of your model and ui-tree in sync.

Example tree data

$scope.treeItems = [
  {
    id: 1,
    title: 'Item 1',
    collapsed: true,
    items: [
      {
      id: 11,
      title: 'Item 1.1'
      }
    ]
  }, {
    id: 2,
    title: 'Item 2',
    items: []
  }
];

Available callbacks

$scope.callbacks = {
    accept: function(sourceNodeScope, destNodesScope, destIndex) {},
    dropped: function(event) {},
    dragStart: function(event) {},
    dragMove: function(event) {},
    dragStop: function(event) {}
};
Select Box

<select hippo.theme.select-box
  multiple="multiple"
  ng-model="selectedValues"
  ng-options="v.name for v in values"
  options="values"
  no-results-text="No results found for"
  placeholder="Please select an item">
</select>

Note: without the 'options' variable, changes to the ng-options attribute value will not be reflected in the UI.
